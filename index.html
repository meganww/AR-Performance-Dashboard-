📄 SQL: AR Yield (COGS-Based) — For Dollar PtP▼
-- AR Yield % = Sellable COGS / Total Graded COGS -- AR Yield PtP = Actual Sellable COGS / OP2 Plan Sellable COGS SELECT TO_CHAR(evaluation_day_lcl, 'YYYY-MM') AS year_month, SUM(cost_sellable) AS total_sellable_cogs, SUM(cost) AS total_graded_cogs, ROUND((SUM(cost_sellable) / NULLIF(SUM(cost), 0)) * 100, 2) AS ar_yield_pct, SUM(ma_op2_sellable_cogs) AS plan_sellable_cogs, ROUND((SUM(cost_sellable) / NULLIF(SUM(ma_op2_sellable_cogs), 0)) * 100, 2) AS ar_yield_ptp FROM andes.rr_metrics.d_aw_yield_ptp WHERE marketplace_id = 1 AND metric = 'aw_yield' AND channel = 'RETAIL' AND evaluation_day_lcl BETWEEN '2026-01-01' AND '2026-03-21' GROUP BY 1 ORDER BY 1;
📄 SQL: AR Yield (Unit-Based) — For Unit PtP▼
-- Unit-based yield works for all channels SELECT TO_CHAR(evaluation_day_lcl, 'YYYY-MM') AS year_month, SUM(sellable_units) AS sellable_units, SUM(sellable_units + unsellable_units) AS total_units, ROUND((SUM(sellable_units) / NULLIF(SUM(sellable_units + unsellable_units), 0)) * 100, 2) AS ar_yield_unit_pct, SUM(ma_op2_sellable_units) AS plan_sellable_units, ROUND((SUM(sellable_units) / NULLIF(SUM(ma_op2_sellable_units), 0)) * 100, 2) AS ar_yield_ptp_unit FROM andes.rr_metrics.d_aw_yield_ptp WHERE marketplace_id = 1 AND metric = 'aw_yield' AND evaluation_day_lcl BETWEEN '2026-01-01' AND '2026-03-21' GROUP BY 1 ORDER BY 1;
📄 SQL: AR Concessions & RRV▼
-- Exclude last 8 weeks for data maturation SELECT TO_CHAR(ship_day, 'YYYY-MM') AS year_month, ROUND((SUM(conc_units_wbr_all) / NULLIF(SUM(ship_units_wbr), 0)) * 100, 4) AS ar_concession_rate_pct, SUM(ship_units_wbr) AS shipped_units, SUM(conc_units_wbr_all) AS conceded_units FROM andes.wdmetrics_serv.ta_whd_concessions WHERE marketplace_id = 1 AND ship_day < DATEADD(week, -8, CURRENT_DATE) AND ship_day >= '2026-01-01' GROUP BY 1 ORDER BY 1;
📄 SQL: Top Opportunity Sites▼
SELECT warehouse_id, SUM(cost_sellable) AS sellable_cogs, SUM(cost) AS graded_cogs, SUM(ma_op2_sellable_cogs) - SUM(cost_sellable) AS opportunity_cogs, ROUND((SUM(cost_sellable) / NULLIF(SUM(cost), 0)) * 100, 2) AS ar_yield_pct, ROUND((SUM(cost_sellable) / NULLIF(SUM(ma_op2_sellable_cogs), 0)) * 100, 2) AS ar_yield_ptp FROM andes.rr_metrics.a_aw_yield_ptp WHERE marketplace_id = '1' AND channel = 'RETAIL' AND evaluation_day_lcl BETWEEN '2026-03-15' AND '2026-03-21' AND relo_ops_region IN ('US Relo - East', 'US Relo - West', 'US Relo - Central') GROUP BY 1 ORDER BY opportunity_cogs DESC LIMIT 10;
